테두리 관련 스타일(border~)

        선택자 {
            * 테두리 스타일
            border[-위치]-style: none | dotted | dashed | solid | groove | ridge | insert | outset;

            * 테두리 두께
            border[-위치]-width: 테두리 두께;

            * 테두리 색상
            border[-위치]-color: 테두리 색상;

            * 테두리를 한꺼번에
            border : 두께 스타일 [색상];
            
            * 테두리 모서리를 둥글게
            border-radius: 숫자(반지름);

            * 박스의 그림자
            box-shadow : 
        }
